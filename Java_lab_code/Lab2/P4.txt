package p4;

import java.util.Scanner;

/**
 *
 * @author royar
 */
public class P4 {

    /**
     * @param args the command line arguments
     */
    public static void main(String[] args) {
          int n;
          Scanner in = new Scanner(System.in);
           System.out.println("enter the n terms of star: ");
            n=in.nextInt();
          for( int i=1; i<=n; i++)
          {
              for(int j=1; j<=n-i; j++)
              {
                  System.out.print(" ");
              }
                  for(int k=1; k<=(2*i-1); k++)
                  {
                      System.out.print("*");
                  }
                      System.out.println("");
            }
    }

    
}
