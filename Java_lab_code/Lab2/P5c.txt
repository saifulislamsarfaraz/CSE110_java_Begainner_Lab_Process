package p5b;

import java.util.Scanner;

/**
 *
 * @author royar
 */
public class P5b {

    /**
     * @param args the command line arguments
     */
    public static void main(String[] args) {
          Scanner in =new Scanner(System.in);
        System.out.println("enter the n terms of star: ");
        int n=in.nextInt();
        for(int i=1;i<=n;i++){
            for(int j=n; j>=i;j--)
            {
                System.out.print(" ");
            }
            for(int k=i;k>=1;k--){
                System.out.print(k+"");
            }
            System.out.println("");
        }  
    }
    
}
